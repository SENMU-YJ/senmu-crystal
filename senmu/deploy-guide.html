部署签名服务（让朋友圈卡片生效的关键）

ℹ️
Vercel 免费版可以部署 Node.js 函数！把 server.js 改造成 Vercel Serverless Function 即可。

在 GitHub 仓库里，新建文件 api/wechat/sign.js（注意路径层级）：

// api/wechat/sign.js ← 文件路径 // Vercel Serverless Function - 签名接口 const axios = require('axios'); const crypto = require('crypto'); // ⬇ 填入你的公众号 AppID 和 AppSecret const APPID = '你的AppID'; const APPSECRET = '你的AppSecret'; // 简单内存缓存（Serverless函数每次可能重启，生产环境建议用Redis） let tokenCache = { value: '', expires: 0 }; let ticketCache = { value: '', expires: 0 }; async function getAccessToken() { if (tokenCache.value && tokenCache.expires > Date.now()) return tokenCache.value; const r = await axios.get( `https://api.weixin.qq.com/cgi-bin/token?grant_type=client_credential&appid=${APPID}&secret=${APPSECRET}` ); tokenCache = { value: r.data.access_token, expires: Date.now() + 7000000 }; return tokenCache.value; } async function getTicket() { if (ticketCache.value && ticketCache.expires > Date.now()) return ticketCache.value; const token = await getAccessToken(); const r = await axios.get( `https://api.weixin.qq.com/cgi-bin/ticket/getticket?access_token=${token}&type=jsapi` ); ticketCache = { value: r.data.ticket, expires: Date.now() + 7000000 }; return ticketCache.value; } module.exports = async (req, res) => { res.setHeader('Access-Control-Allow-Origin', '*'); const url = req.query.url || ''; const ticket = await getTicket(); const timestamp = Math.floor(Date.now() / 1000); const nonceStr = Math.random().toString(36).substring(2, 17); const signStr = `jsapi_ticket=${ticket}&noncestr=${nonceStr}&timestamp=${timestamp}&url=${url}`; const signature = crypto.createHash('sha1').update(signStr).digest('hex'); res.json({ appId: APPID, timestamp, nonceStr, signature }); };

同时修改 package.json，加入 axios 依赖：

{ "name": "senmu-crystal", "dependencies": { "axios": "^1.6.0" } }
✅
文件上传到 GitHub 后，Vercel 自动检测 api/ 目录为 Serverless 函数，无需额外配置！
